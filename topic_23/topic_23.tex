% ===== SECTION 1: MASS-ENERGY EQUIVALENCE =====
\section{Mass--Energy Equivalence}

\begin{definitionbox}{Einstein's Mass--Energy Relation}
Einstein's special theory of relativity establishes that mass and energy are equivalent. A mass $m$ at rest has an intrinsic energy given by:
$$E = mc^2$$
\begin{itemize}[itemsep=2pt]
  \item $c = 3.00 \times 10^8\ \text{m s}^{-1}$ (speed of light in free space)
  \item A small mass corresponds to an enormous amount of energy.
  \item In nuclear reactions, small changes in mass $\Delta m$ release measurable amounts of energy.
\end{itemize}
\end{definitionbox}

\begin{definitionbox}{Atomic Mass Unit}
The \textbf{unified atomic mass unit} (u) is defined as one-twelfth of the mass of a carbon-12 atom.
$$1\ \text{u} = 1.661 \times 10^{-27}\ \text{kg}$$
The energy equivalent of 1 u:
$$E = mc^2 = 1.661\times10^{-27} \times (3.00\times10^8)^2 = 1.49\times10^{-10}\ \text{J} = 931.5\ \text{MeV}$$
So $1\ \text{u} \equiv 931.5\ \text{MeV}/c^2$.
\end{definitionbox}

\begin{keybox}{Nuclear Notation and Equations}
A nuclide is written $^A_Z\text{X}$, where $A$ is the \textbf{nucleon number} (mass number) and $Z$ is the \textbf{proton number} (atomic number).\\[0.5em]
Nuclear equations must conserve:
\begin{itemize}[itemsep=2pt]
  \item \textbf{Nucleon number} $A$ (top numbers balance)
  \item \textbf{Proton number} $Z$ (bottom numbers balance)
  \item \textbf{Mass--energy} (energy is released or absorbed)
  \item \textbf{Charge} and \textbf{momentum}
\end{itemize}
Example: $^{14}_{7}\text{N} + ^{4}_{2}\text{He} \rightarrow ^{17}_{8}\text{O} + ^{1}_{1}\text{H}$
\end{keybox}

% ===== SECTION 2: MASS DEFECT AND BINDING ENERGY =====
\section{Mass Defect and Binding Energy}

\begin{definitionbox}{Mass Defect}
The \textbf{mass defect} $\Delta m$ of a nucleus is the difference between the total mass of the separate constituent nucleons and the actual mass of the nucleus.
$$\Delta m = Z\,m_p + (A-Z)\,m_n - m_{\text{nucleus}}$$
\begin{tabular}{@{}ll@{}}
$m_p$ & = mass of a proton $= 1.6726\times10^{-27}\ \text{kg}$ \\
$m_n$ & = mass of a neutron $= 1.6749\times10^{-27}\ \text{kg}$ \\
$m_{\text{nucleus}}$ & = actual measured mass of the nucleus \\
\end{tabular}\\[0.4em]
The mass defect is always \textbf{positive}: the nucleus is always less massive than its parts.
\end{definitionbox}

\begin{definitionbox}{Binding Energy}
The \textbf{binding energy} of a nucleus is the energy required to completely separate a nucleus into its constituent protons and neutrons (i.e.\ to infinity).
$$E_B = \Delta m \cdot c^2$$
Equivalently, it is the energy \emph{released} when the nucleus is assembled from separate nucleons.
\end{definitionbox}

\begin{formulabox}{Energy Released in a Nuclear Reaction}
\begin{center}
\Large $E = c^2\,\Delta m$
\end{center}
\vspace{0.3em}
where $\Delta m$ is the difference between the total mass of reactants and the total mass of products.\\
If $\Delta m > 0$ (reactants heavier than products): energy is \textbf{released}.\\
If $\Delta m < 0$: energy must be \textbf{supplied}.
\end{formulabox}

\begin{warningbox}{Common Mistake — Mass Defect vs Binding Energy}
Students often confuse the \emph{sign convention}. The mass defect is always defined as a positive quantity (how much mass is ``missing''). The binding energy is the energy equivalent of this missing mass. A \textbf{larger} binding energy means a \textbf{more stable} nucleus — it takes more energy to pull it apart.
\end{warningbox}

% ===== SECTION 3: BINDING ENERGY PER NUCLEON =====
\section{Binding Energy per Nucleon}

\begin{definitionbox}{Binding Energy per Nucleon}
The \textbf{binding energy per nucleon} is the total binding energy of a nucleus divided by its nucleon number $A$. It is a measure of nuclear stability: the higher the value, the more stable the nucleus.
$$\text{BE per nucleon} = \frac{E_B}{A} = \frac{c^2 \Delta m}{A}$$
\end{definitionbox}

% ===== SECTION 4: FISSION AND FUSION =====
\section{Nuclear Fission and Fusion}

\begin{definitionbox}{Nuclear Fission}
\textbf{Nuclear fission} is the splitting of a large, unstable nucleus into two smaller (daughter) nuclei of roughly equal mass, accompanied by the release of neutrons and energy.
\begin{itemize}[itemsep=2pt]
  \item Induced fission: a slow (thermal) neutron is absorbed by a heavy nucleus (e.g.\ $^{235}$U), which then splits.
  \item The products have greater binding energy per nucleon than the original nucleus $\Rightarrow$ energy is released.
  \item Typically releases 2--3 fast neutrons which can trigger further fissions (\textbf{chain reaction}).
\end{itemize}
Example: $^{235}_{92}\text{U} + ^{1}_{0}\text{n} \rightarrow ^{141}_{56}\text{Ba} + ^{92}_{36}\text{Kr} + 3\,^{1}_{0}\text{n}$
\end{definitionbox}

\begin{definitionbox}{Nuclear Fusion}
\textbf{Nuclear fusion} is the combining of two light nuclei to form a heavier nucleus, releasing energy.
\begin{itemize}[itemsep=2pt]
  \item The product nucleus has greater binding energy per nucleon than the reactants $\Rightarrow$ energy is released.
  \item Requires \textbf{extremely high temperatures} ($\sim 10^7\ \text{K}$) to overcome electrostatic repulsion between nuclei.
  \item Powers stars; the basis of proposed fusion reactors.
\end{itemize}
Example: $^{2}_{1}\text{H} + ^{3}_{1}\text{H} \rightarrow ^{4}_{2}\text{He} + ^{1}_{0}\text{n}$
\end{definitionbox}

\begin{keybox}{Why These Reactions Release Energy}
In both fission and fusion, the total mass of the \textbf{products} is less than the total mass of the \textbf{reactants}. This mass difference $\Delta m$ is converted to kinetic energy of the products via $E = c^2\Delta m$. The reactions move nuclei \emph{towards} the peak of the BE per nucleon curve (towards $^{56}$Fe).
\end{keybox}

% ===== SECTION 5: RADIOACTIVE DECAY =====
\section{Radioactive Decay}

\begin{definitionbox}{Spontaneous and Random Decay}
\textbf{Radioactive decay} is the spontaneous emission of radiation from an unstable nucleus.
\begin{itemize}[itemsep=2pt]
  \item \textbf{Spontaneous}: the decay is not triggered by external conditions (temperature, pressure, chemical state); it cannot be predicted or controlled.
  \item \textbf{Random}: it is impossible to predict \emph{when} any particular nucleus will decay. Each nucleus has the same probability of decaying per unit time.
  \item \textbf{Evidence for randomness}: fluctuations (statistical variation) in the measured count rate from a radioactive source.
\end{itemize}
\end{definitionbox}

\begin{definitionbox}{Activity and Decay Constant}
The \textbf{activity} $A$ of a source is the number of nuclei that decay per unit time.
$$A = \lambda N$$
\begin{tabular}{@{}ll@{}}
$A$ & = activity (Bq, where $1\ \text{Bq} = 1\ \text{decay s}^{-1}$) \\
$\lambda$ & = \textbf{decay constant} — the probability of decay of a nucleus per unit time (s$^{-1}$) \\
$N$ & = number of undecayed nuclei present \\
\end{tabular}
\end{definitionbox}

\begin{definitionbox}{Half-Life}
The \textbf{half-life} $t_{1/2}$ is the time taken for the number of undecayed nuclei (or the activity) of a radioactive sample to fall to half its initial value.
$$\lambda = \frac{\ln 2}{t_{1/2}} = \frac{0.693}{t_{1/2}}$$
Half-life is constant for a given isotope — it does not depend on the number of nuclei present or external conditions.
\end{definitionbox}

% ===== SECTION 6: EXPONENTIAL DECAY =====
\section{Exponential Decay}

\begin{formulabox}{Exponential Decay Equations}
\begin{center}
\Large $x = x_0\,e^{-\lambda t}$
\end{center}
\vspace{0.3em}
where $x$ can represent:
\begin{itemize}[itemsep=2pt]
  \item $N$ = number of undecayed nuclei: \quad $N = N_0\,e^{-\lambda t}$
  \item $A$ = activity of the source: \quad $A = A_0\,e^{-\lambda t}$
  \item $C$ = received count rate: \quad $C = C_0\,e^{-\lambda t}$
\end{itemize}
\vspace{0.4em}
\begin{tabular}{@{}ll@{}}
$x_0$ & = initial value at $t = 0$ \\
$\lambda$ & = decay constant (s$^{-1}$) \\
$t$ & = time elapsed (s) \\
\end{tabular}
\end{formulabox}

\begin{keybox}{Linearising the Decay Equation}
Taking the natural logarithm of $N = N_0 e^{-\lambda t}$:
$$\ln N = \ln N_0 - \lambda t$$
A graph of $\ln N$ (or $\ln A$) against $t$ gives a \textbf{straight line} with:
\begin{itemize}[itemsep=2pt]
  \item Gradient $= -\lambda$
  \item y-intercept $= \ln N_0$
\end{itemize}
This is the standard experimental method to determine $\lambda$ and hence $t_{1/2}$.
\end{keybox}

\begin{warningbox}{Common Errors with Decay Calculations}
\begin{itemize}[itemsep=3pt]
  \item Using $t_{1/2}$ directly in the exponential formula — you must use $\lambda$, not $t_{1/2}$. Convert first: $\lambda = 0.693/t_{1/2}$.
  \item Forgetting to convert time units: if $t_{1/2}$ is in days, convert to seconds before finding $\lambda$ in s$^{-1}$.
  \item Confusing \textbf{activity} $A = \lambda N$ (Bq) with \textbf{count rate} — the count rate is always less than activity due to detector efficiency and geometry.
  \item Applying the exponential formula to something that \emph{increases} over time — it only applies to $N$, $A$, or count rate, which all decay.
\end{itemize}
\end{warningbox}

% ===== SECTION 8: EXAM TECHNIQUE =====
\section{Exam Technique and Problem-Solving Strategy}

\begin{keybox}{Step-by-Step Strategy for Nuclear Calculations}
\begin{enumerate}[itemsep=3pt]
  \item \textbf{Balance the equation} — check nucleon numbers and proton numbers sum correctly on both sides.
  \item \textbf{Find} $\Delta m$ — total reactant mass minus total product mass; work in kg or u.
  \item \textbf{Apply} $E = c^2\Delta m$ to find energy released (convert u to kg if needed, or use $1\ \text{u} = 931.5\ \text{MeV}$).
  \item For decay: \textbf{convert} $t_{1/2}$ to seconds, find $\lambda = 0.693/t_{1/2}$, then apply $x = x_0 e^{-\lambda t}$.
\end{enumerate}
\end{keybox}

\begin{warningbox}{Common Errors — Avoid These!}
\begin{itemize}[itemsep=3pt]
  \item Using \textbf{atomic masses} (including electron masses) rather than nuclear masses without accounting for electron masses — take care with data provided in exam questions.
  \item Forgetting that $\Delta m$ must be in \textbf{kg} when using $E = mc^2$ in SI units.
  \item Confusing \textbf{binding energy} with \textbf{ionisation energy} — binding energy refers to the nucleus.
  \item Stating that a higher binding energy per nucleon means \textbf{less} stable — it means \textbf{more} stable.
  \item Mixing up the direction of fusion and fission on the BE/nucleon graph.
\end{itemize}
\end{warningbox}

% ===== SECTION 9: WORKED EXAMPLES =====
\section{Worked Examples}

\subsection*{Example 1 — Mass Defect and Binding Energy}

\textbf{Question:} Calculate the mass defect and binding energy of a helium-4 nucleus ($^4_2$He). \\
($m_p = 1.6726\times10^{-27}\ \text{kg}$, $m_n = 1.6749\times10^{-27}\ \text{kg}$, $m_{^4\text{He}} = 6.6447\times10^{-27}\ \text{kg}$)

\begin{examplebox}{Solution}
\textbf{Solution:}\\
Mass of constituents: $2m_p + 2m_n = 2(1.6726) + 2(1.6749) = 6.6950\times10^{-27}\ \text{kg}$

Mass defect:
$$\Delta m = 6.6950\times10^{-27} - 6.6447\times10^{-27} = \mathbf{5.03\times10^{-29}\ \text{kg}}$$

Binding energy:
$$E_B = \Delta m\,c^2 = 5.03\times10^{-29} \times (3.00\times10^8)^2 = \mathbf{4.53\times10^{-12}\ \text{J}}\ (= 28.3\ \text{MeV})$$

Binding energy per nucleon: $4.53\times10^{-12}/4 = 1.13\times10^{-12}\ \text{J} = 7.07\ \text{MeV per nucleon}$
\end{examplebox}

\subsection*{Example 2 — Energy Released in Fission}

\textbf{Question:} In a fission reaction, the total mass of products is $3.09\times10^{-28}\ \text{kg}$ less than the total mass of reactants. Calculate the energy released in MeV.

\begin{examplebox}{Solution}
\textbf{Solution:}\\
$$E = c^2\Delta m = (3.00\times10^8)^2 \times 3.09\times10^{-28} = 2.78\times10^{-11}\ \text{J}$$
$$E = \frac{2.78\times10^{-11}}{1.60\times10^{-13}} = \mathbf{174\ \text{MeV}}$$
\end{examplebox}

\subsection*{Example 3 — Radioactive Decay Calculation}

\textbf{Question:} A radioactive isotope has a half-life of $12.0$ hours. A sample initially contains $8.00\times10^{20}$ undecayed nuclei. Calculate (a) the decay constant, (b) the initial activity, (c) the number of undecayed nuclei after 30.0 hours.

\begin{examplebox}{Solution}
\textbf{Solution:}\\
(a) Convert: $t_{1/2} = 12.0 \times 3600 = 4.32\times10^4\ \text{s}$
$$\lambda = \frac{0.693}{4.32\times10^4} = \mathbf{1.60\times10^{-5}\ \text{s}^{-1}}$$

(b) $A_0 = \lambda N_0 = 1.60\times10^{-5} \times 8.00\times10^{20} = \mathbf{1.28\times10^{16}\ \text{Bq}}$

(c) $t = 30.0 \times 3600 = 1.08\times10^5\ \text{s}$
$$N = N_0\,e^{-\lambda t} = 8.00\times10^{20} \times e^{-1.60\times10^{-5}\times1.08\times10^5}$$
$$N = 8.00\times10^{20} \times e^{-1.728} = 8.00\times10^{20} \times 0.178 = \mathbf{1.42\times10^{20}}$$
\end{examplebox}

\section{Mark Scheme and Answers}

\begin{markscheme}

\textbf{Q1.} \textit{Mass defect}: the difference between the total mass of the separate nucleons (protons and neutrons) and the actual mass of the nucleus [2]. \textit{Binding energy}: the energy required to completely separate a nucleus into its constituent protons and neutrons [2].

\vspace{0.5em}\textbf{Q2.} The curve rises steeply for light nuclei — fusion of light nuclei produces a product with greater BE/nucleon, so energy is released [1]. The curve falls for heavy nuclei — fission of a heavy nucleus produces fragments with greater BE/nucleon, so energy is released [1].

\vspace{0.5em}\textbf{Q3.} $N = A/\lambda = 6.4\times10^5 / 2.0\times10^{-3} = \mathbf{3.2\times10^8}$ [2]. $t_{1/2} = 0.693/\lambda = 0.693/2.0\times10^{-3} = \mathbf{347\ \text{s}}$ [2].

\vspace{0.5em}\textbf{Q4.} \textit{Spontaneous}: the decay is not triggered or affected by external conditions; it cannot be induced or prevented [1]. \textit{Random}: it is impossible to predict which nucleus will decay next, or when; each nucleus has the same fixed probability of decaying per unit time [1].

\vspace{0.5em}\textbf{Q5.} Nucleon: $238 = 234 + A \Rightarrow A = 4$; proton: $92 = 90 + Z \Rightarrow Z = 2$. X is $^4_2\text{He}$ (an alpha particle) [2].

\vspace{0.5em}\textbf{Q6(a).} $\Delta m = (2.01410 + 3.01605) - (4.00260 + 1.00867) = 5.03015 - 5.01127 = 0.01888\ \text{u}$ [1]; $= 0.01888 \times 1.661\times10^{-27} = \mathbf{3.14\times10^{-29}\ \text{kg}}$ [2].

\vspace{0.5em}\textbf{Q6(b).} $E = c^2\Delta m = (3.00\times10^8)^2 \times 3.14\times10^{-29} = \mathbf{2.82\times10^{-12}\ \text{J}}$ [1]; $= 2.82\times10^{-12}/1.60\times10^{-13} = \mathbf{17.6\ \text{MeV}}$ [1].

\vspace{0.5em}\textbf{Q6(c).} The reactants ($^2$H and $^3$H) lie to the left of the peak of the BE/nucleon curve [1]; the product $^4$He has a higher binding energy per nucleon than the reactants [1]; since the products are more tightly bound, mass is converted to energy and released [1].

\vspace{0.5em}\textbf{Q7(a).} $t_{1/2} = 8.04 \times 24 \times 3600 = 6.95\times10^5\ \text{s}$; $\lambda = 0.693/6.95\times10^5 = \mathbf{9.97\times10^{-7}\ \text{s}^{-1}}$ [2].

\vspace{0.5em}\textbf{Q7(b).} 24 days $= 3 \times t_{1/2}$; $A = 4.0\times10^8 \times (1/2)^3 = 4.0\times10^8/8 = \mathbf{5.0\times10^7\ \text{Bq}}$ [2].

\vspace{0.5em}\textbf{Q7(c).} Straight line [1]; y-intercept at $\ln(4.0\times10^8) = 19.8$ labelled $\ln A_0$ [1]; gradient $= -\lambda$ (negative slope) [1].
