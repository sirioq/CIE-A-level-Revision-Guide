% ===== SECTION 1: KEY CONCEPTS =====
\section{Core Concepts and Definitions}

\begin{definitionbox}{The Quantum Model of Light}
Classical wave theory cannot explain certain phenomena involving light and matter. Quantum theory proposes that electromagnetic radiation is emitted and absorbed in discrete packets of energy called \textbf{photons}.
\begin{itemize}[itemsep=2pt]
  \item A \textbf{photon} is a quantum of electromagnetic energy.
  \item The energy of a photon depends only on its \textbf{frequency}, not its intensity.
  \item At a fixed frequency, intensity is proportional to the \textbf{number of photons} per unit time. Increasing intensity does \textbf{not} increase the energy of individual photons.
\end{itemize}
\end{definitionbox}

\begin{formulabox}{Photon Energy — The Planck Relation}
\begin{center}
\Large $E = hf = \dfrac{hc}{\lambda}$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$E$ & = energy of one photon (J) \\
$h$ & = Planck's constant $= 6.63 \times 10^{-34}\ \text{J s}$ \\
$f$ & = frequency of the radiation (Hz) \\
$c$ & = speed of light $= 3.00 \times 10^{8}\ \text{m s}^{-1}$ \\
$\lambda$ & = wavelength (m) \\
\end{tabular}
\end{formulabox}

\begin{keybox}{The Electronvolt}
The \textbf{electronvolt} (eV) is a convenient unit of energy at the atomic scale.
$$1\ \text{eV} = 1.60 \times 10^{-19}\ \text{J}$$
To convert eV $\to$ J: multiply by $1.60\times10^{-19}$.\\
To convert J $\to$ eV: divide by $1.60\times10^{-19}$.
\end{keybox}

% ===== SECTION 2: PHOTOELECTRIC EFFECT =====
\section{The Photoelectric Effect}

\begin{definitionbox}{The Photoelectric Effect}
The \textbf{photoelectric effect} is the emission of electrons from a metal surface when electromagnetic radiation of sufficiently high frequency is incident on it.
\begin{itemize}[itemsep=2pt]
  \item Electrons emitted are called \textbf{photoelectrons}.
  \item Emission occurs \textbf{instantaneously} if the frequency is above a threshold — there is no time delay.
  \item Below the \textbf{threshold frequency} $f_0$, no electrons are emitted regardless of intensity.
  \item Above $f_0$, increasing the intensity increases the \textbf{number} of photoelectrons per second, not their energy.
\end{itemize}
\end{definitionbox}

\begin{warningbox}{Why Classical Wave Theory Fails}
\begin{itemize}[itemsep=3pt]
  \item Wave theory predicts that any frequency of sufficient intensity should eventually eject electrons — \textbf{not observed}.
  \item Wave theory predicts a time delay before emission as energy builds up — \textbf{not observed} (emission is instantaneous).
  \item Wave theory predicts that intensity should increase the electron's kinetic energy — \textbf{not observed}.
\end{itemize}
These failures led Einstein to propose the photon model.
\end{warningbox}

\begin{formulabox}{Einstein's Photoelectric Equation}
\begin{center}
\Large $hf = \phi + \tfrac{1}{2}mv^2_{\max}$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$hf$ & = energy of the incident photon (J) \\
$\phi$ & = \textbf{work function} — minimum energy to remove an electron from the surface (J) \\
$\tfrac{1}{2}mv^2_{\max}$ & = maximum kinetic energy of emitted photoelectron (J) \\
\end{tabular}
\vspace{0.5em}

The \textbf{work function} $\phi = hf_0$, where $f_0$ is the threshold frequency.\\
Maximum KE can also be written as $eV_s$, where $V_s$ is the \textbf{stopping potential}.
\end{formulabox}

\begin{definitionbox}{Stopping Potential $V_s$}
The \textbf{stopping potential} is the minimum opposing potential difference required to prevent all photoelectrons from reaching the collecting electrode.
$$eV_s = \tfrac{1}{2}mv^2_{\max} = hf - \phi$$
Measuring $V_s$ at different frequencies allows $h$ and $\phi$ to be determined experimentally.
\end{definitionbox}

% ===== SECTION 3: WAVE-PARTICLE DUALITY =====
\section{Wave--Particle Duality}

\begin{definitionbox}{Wave--Particle Duality}
\textbf{Wave--particle duality} is the property of matter and radiation by which they exhibit both wave-like and particle-like behaviour depending on the experimental context.
\begin{itemize}[itemsep=2pt]
  \item Light behaves as a \textbf{wave}: diffraction, interference, polarisation.
  \item Light behaves as \textbf{particles (photons)}: photoelectric effect, Compton scattering.
  \item Electrons (and other particles) also show \textbf{wave behaviour}: electron diffraction.
\end{itemize}
\end{definitionbox}

\begin{formulabox}{de Broglie Wavelength}
\begin{center}
\Large $\lambda = \dfrac{h}{p} = \dfrac{h}{mv}$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$\lambda$ & = de Broglie wavelength (m) \\
$h$ & = Planck's constant ($6.63 \times 10^{-34}$ J s) \\
$p$ & = momentum of the particle (kg m s$^{-1}$) \\
$m$ & = mass of the particle (kg) \\
$v$ & = speed of the particle (m s$^{-1}$) \\
\end{tabular}
\end{formulabox}

\begin{keybox}{Electron Diffraction}
When electrons are accelerated through a potential difference $V$ and directed at a thin crystal or graphite film, a \textbf{diffraction pattern} of rings is observed — confirming their wave nature.
\begin{itemize}[itemsep=2pt]
  \item The kinetic energy gained: $\tfrac{1}{2}mv^2 = eV$, giving $v = \sqrt{2eV/m}$.
  \item Therefore: $\lambda = \dfrac{h}{\sqrt{2meV}}$
  \item Increasing $V$ increases $p$, decreasing $\lambda$ $\Rightarrow$ rings become \textbf{smaller}.
  \item The wavelength must be comparable to the atomic spacing for diffraction to occur ($\sim 10^{-10}$ m).
\end{itemize}
\end{keybox}

\begin{warningbox}{Common Mistake — Confusing Photon and Particle Momentum}
For a \textbf{photon} (massless): $p = E/c = hf/c = h/\lambda$.\\
For a \textbf{particle with mass}: $p = mv = h/\lambda$ (de Broglie).\\
Both use $\lambda = h/p$, but the relationship between energy and momentum differs. Do not use $E = hf$ for material particles.
\end{warningbox}

% ===== SECTION 4: ENERGY LEVELS AND LINE SPECTRA =====
\section{Energy Levels and Line Spectra}

\begin{definitionbox}{Atomic Energy Levels}
Electrons in an atom can only occupy discrete \textbf{energy levels}. These are quantised — only specific energies are allowed.
\begin{itemize}[itemsep=2pt]
  \item The \textbf{ground state} is the lowest energy level (most stable; most negative value).
  \item \textbf{Excited states} have higher (less negative) energy.
  \item The \textbf{ionisation energy} is the energy required to remove an electron from the ground state to infinity ($E = 0$).
  \item Energy levels are usually expressed in electronvolts (eV) and are \textbf{negative} (bound states).
\end{itemize}
\end{definitionbox}

\begin{formulabox}{Photon Energy from a Transition}
When an electron moves between energy levels $E_1$ (lower) and $E_2$ (higher):
\begin{center}
\Large $hf = E_2 - E_1$
\end{center}
\begin{itemize}[itemsep=2pt]
  \item \textbf{Emission}: electron falls to lower level $\Rightarrow$ photon \textbf{emitted} with $hf = E_2 - E_1$.
  \item \textbf{Absorption}: electron promoted to higher level $\Rightarrow$ photon \textbf{absorbed} with $hf = E_2 - E_1$.
\end{itemize}
\end{formulabox}

\subsection{Emission and Absorption Spectra}

\begin{definitionbox}{Line Spectra}
\begin{itemize}[itemsep=4pt]
  \item \textbf{Emission spectrum}: a series of \textbf{bright coloured lines} on a dark background. Each line corresponds to a specific photon frequency emitted during a downward electron transition. Produced by excited gases.
  \item \textbf{Absorption spectrum}: a \textbf{continuous spectrum} crossed by dark lines at the same frequencies as the emission lines. Cool gas in front of a broad-spectrum source absorbs specific photon energies.
  \item The \textbf{line positions are unique} to each element — used for identification (spectroscopy).
\end{itemize}
\end{definitionbox}

\begin{keybox}{Why Line Spectra Prove Quantisation}
Because electrons can only occupy discrete energy levels, only photons of specific energies (frequencies) can be absorbed or emitted. This produces a \textbf{line} spectrum rather than a continuous one. A continuous spectrum would imply electrons can have any energy — which they cannot.
\end{keybox}

% ===== SECTION 7: EXAM TECHNIQUE =====
\section{Exam Technique and Problem-Solving Strategy}

\begin{keybox}{Step-by-Step Strategy for Calculation Questions}
\begin{enumerate}[itemsep=3pt]
  \item \textbf{Identify} the relevant formula — is it a photon, a particle, or a transition?
  \item \textbf{Convert units}: eV to J ($\times 1.60\times10^{-19}$); nm to m ($\times 10^{-9}$).
  \item \textbf{Substitute} values carefully, showing all working.
  \item \textbf{Quote} the final answer in appropriate units with correct significant figures.
\end{enumerate}
\end{keybox}

\begin{warningbox}{Common Errors — Avoid These!}
\begin{itemize}[itemsep=3pt]
  \item Using the \textbf{wrong energy unit} — forgetting to convert eV to J before substituting.
  \item Confusing \textbf{threshold frequency} $f_0$ with threshold wavelength: a higher $f_0$ corresponds to a \textbf{shorter} threshold wavelength $\lambda_0$.
  \item Applying \textbf{intensity} changes to change photon energy — intensity only changes the \textbf{number} of photons.
  \item Using $E = hf$ for the de Broglie relation — this is for photons only; for matter use $\lambda = h/mv$.
  \item Forgetting that energy levels are \textbf{negative} — the transition energy is $|E_2 - E_1|$, not just $E_2 - E_1$ as magnitudes.
  \item Confusing \textbf{emission} (bright lines) and \textbf{absorption} (dark lines) spectra.
\end{itemize}
\end{warningbox}

% ===== SECTION 8: WORKED EXAMPLES =====
\section{Worked Examples}

\subsection*{Example 1 — Photoelectric Effect Calculation}

\textbf{Question:} Light of wavelength $250\ \text{nm}$ is incident on a metal surface with a work function of $4.5\ \text{eV}$. Calculate the maximum kinetic energy of emitted photoelectrons and the stopping potential.

\begin{examplebox}{Solution}
\textbf{Solution:}\\
Convert work function: $\phi = 4.5 \times 1.60\times10^{-19} = 7.20\times10^{-19}\ \text{J}$

Photon energy:
$$E = \frac{hc}{\lambda} = \frac{6.63\times10^{-34} \times 3.00\times10^{8}}{250\times10^{-9}} = 7.96\times10^{-19}\ \text{J}$$

Maximum KE:
$$E_{k(\max)} = hf - \phi = 7.96\times10^{-19} - 7.20\times10^{-19} = \mathbf{7.6\times10^{-20}\ \text{J}}$$

Stopping potential:
$$V_s = \frac{E_{k(\max)}}{e} = \frac{7.6\times10^{-20}}{1.60\times10^{-19}} = \mathbf{0.48\ \text{V}}$$
\end{examplebox}

\subsection*{Example 2 — de Broglie Wavelength}

\textbf{Question:} An electron is accelerated from rest through a potential difference of $3.0\ \text{kV}$. Calculate its de Broglie wavelength.

\begin{examplebox}{Solution}
\textbf{Solution:}\\
Energy gained: $E_k = eV = 1.60\times10^{-19} \times 3000 = 4.80\times10^{-16}\ \text{J}$

Speed: $v = \sqrt{\dfrac{2E_k}{m_e}} = \sqrt{\dfrac{2\times4.80\times10^{-16}}{9.11\times10^{-31}}} = 3.25\times10^7\ \text{m s}^{-1}$

de Broglie wavelength:
$$\lambda = \frac{h}{m_e v} = \frac{6.63\times10^{-34}}{9.11\times10^{-31}\times 3.25\times10^7} = \mathbf{2.24\times10^{-11}\ \text{m}}$$

This is comparable to atomic spacings ($\sim 10^{-10}$ m), confirming why electron diffraction is observed.
\end{examplebox}

\subsection*{Example 3 — Energy Level Transition}

\textbf{Question:} An electron in a hydrogen-like atom falls from an energy level of $-1.5\ \text{eV}$ to $-3.4\ \text{eV}$. Find the frequency and wavelength of the emitted photon.

\begin{examplebox}{Solution}
\textbf{Solution:}\\
Energy of emitted photon:
$$hf = E_2 - E_1 = -1.5 - (-3.4) = 1.9\ \text{eV} = 1.9\times1.60\times10^{-19} = 3.04\times10^{-19}\ \text{J}$$

Frequency:
$$f = \frac{3.04\times10^{-19}}{6.63\times10^{-34}} = \mathbf{4.59\times10^{14}\ \text{Hz}} \quad \text{(visible light — red/orange)}$$

Wavelength:
$$\lambda = \frac{c}{f} = \frac{3.00\times10^8}{4.59\times10^{14}} = \mathbf{654\ \text{nm}}$$
\end{examplebox}

% ===== SECTION 10: MARK SCHEME =====
\section{Mark Scheme and Answers}

\begin{markscheme}

\textbf{Q1.} The threshold frequency is the minimum frequency of electromagnetic radiation [1] that is capable of causing photoelectric emission from the surface [1].

\vspace{0.5em}\textbf{Q2.} Any two from: wave theory predicts a time delay before emission, but emission is instantaneous [1]; wave theory predicts any frequency of sufficient intensity should cause emission, but no emission occurs below $f_0$ regardless of intensity [1]; wave theory predicts intensity should increase electron KE, but maximum KE depends only on frequency [1].

\vspace{0.5em}\textbf{Q3.} Photon energy: $E = hf = 6.63\times10^{-34}\times8.0\times10^{14} = 5.30\times10^{-19}\ \text{J} = 3.32\ \text{eV}$ [1]. Since $3.32\ \text{eV} > 2.5\ \text{eV}$, emission \textbf{will} occur [1]. Maximum KE $= 3.32 - 2.5 = 0.82\ \text{eV} = 1.31\times10^{-19}\ \text{J}$ [2].

\vspace{0.5em}\textbf{Q4.} $\lambda = h/(m_p v) = 6.63\times10^{-34}/(1.67\times10^{-27}\times2.0\times10^6) = \mathbf{1.98\times10^{-13}\ \text{m}}$ [2].

\vspace{0.5em}\textbf{Q5.} Both particles gain the same kinetic energy $eV$, so $\tfrac{1}{2}mv^2 = eV \Rightarrow p = mv = \sqrt{2meV}$ [1]; therefore $\lambda = h/\sqrt{2meV}$ [1]; since $m_e \ll m_p$, the denominator is smaller for the electron, giving a \textbf{larger} $\lambda$ for the electron [1].

\vspace{0.5em}\textbf{Q6(a).} $\phi = hf_0 = 6.63\times10^{-34}\times5.5\times10^{14} = \mathbf{3.65\times10^{-19}\ \text{J}}$ [2].

\vspace{0.5em}\textbf{Q6(b).} $E_{k(\max)} = hf - \phi = 6.63\times10^{-34}\times9.0\times10^{14} - 3.65\times10^{-19} = 5.97\times10^{-19} - 3.65\times10^{-19} = 2.32\times10^{-19}\ \text{J}$ [2]; $V_s = E_{k(\max)}/e = 2.32\times10^{-19}/1.60\times10^{-19} = \mathbf{1.45\ \text{V}}$ [1].

\vspace{0.5em}\textbf{Q6(c).} (i) Rate of electron emission \textbf{doubles} [1]: greater intensity $\Rightarrow$ more photons per second $\Rightarrow$ more photoelectron emissions per second [1]. (ii) Maximum KE is \textbf{unchanged} [1]: each photon still has the same energy ($f$ unchanged); intensity affects number of photons, not energy per photon [1].

\vspace{0.5em}\textbf{Q7(a).} $\Delta E = -1.5 - (-13.6) = 12.1\ \text{eV} = 12.1\times1.60\times10^{-19} = 1.936\times10^{-18}\ \text{J}$ [1]; $f = \Delta E/h = 1.936\times10^{-18}/6.63\times10^{-34} = \mathbf{2.92\times10^{15}\ \text{Hz}}$ (ultraviolet) [2].

\vspace{0.5em}\textbf{Q7(b).} The energy difference from $n=1$ to $n=3$ is $-1.5 - (-13.6) = 12.1\ \text{eV}$ [1]; the photon energy exactly matches this transition, so it \textbf{will} be absorbed and the electron promoted to $n=3$ [1].

\vspace{0.5em}\textbf{Q7(c).} Each line in the emission spectrum corresponds to a specific photon frequency [1]; each frequency corresponds to a specific photon energy, given by $hf = E_2 - E_1$ [1]; the existence of discrete lines (rather than a continuum) shows that electrons can only transition between fixed energy values — i.e.\ energy levels are quantised [1].

\end{markscheme}
