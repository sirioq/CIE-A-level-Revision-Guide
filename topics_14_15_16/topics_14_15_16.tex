% ===== SECTION 1: THERMAL EQUILIBRIUM =====
\section{Thermal Equilibrium}

\begin{definitionbox}{Thermal Energy Transfer}
Thermal (heat) energy is transferred from a region of \textbf{higher temperature} to a region of \textbf{lower temperature}. Transfer continues until the two regions reach the same temperature.
\end{definitionbox}

\begin{definitionbox}{Thermal Equilibrium}
Two objects are in \textbf{thermal equilibrium} when there is no net transfer of thermal energy between them. This occurs when they are at the \textbf{same temperature}.

\medskip
\textit{Zeroth Law of Thermodynamics:} If object A is in thermal equilibrium with object C, and object B is also in thermal equilibrium with object C, then A and B are in thermal equilibrium with each other.
\end{definitionbox}

% ===== SECTION 2: TEMPERATURE SCALES =====
\section{Temperature Scales}

\begin{definitionbox}{Thermometric Property}
A \textbf{thermometric property} is a physical property that varies continuously and measurably with temperature. Examples include:
\begin{itemize}[itemsep=2pt]
  \item Resistance of a metal wire (e.g.\ platinum resistance thermometer)
  \item e.m.f.\ of a thermocouple
  \item Volume of a gas at constant pressure
  \item Density of a liquid
\end{itemize}
\end{definitionbox}

\begin{definitionbox}{Thermodynamic Temperature Scale}
The \textbf{thermodynamic (Kelvin) temperature scale} does not depend on the property of any particular substance. It is an absolute scale with:
\begin{itemize}[itemsep=2pt]
  \item \textbf{Absolute zero} ($0\ \text{K}$) --- the lowest possible temperature
  \item \textbf{Triple point of water} ($273.16\ \text{K}$) --- fixed reference point
\end{itemize}
\end{definitionbox}

\begin{formulabox}{Temperature Conversion}
\begin{center}
\Large $T/\text{K} = \theta/\text{°C} + 273.15$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$T$ & = thermodynamic temperature (K) \\
$\theta$ & = Celsius temperature (°C) \\
\end{tabular}

\medskip
Note: a temperature \emph{difference} of $1\ \text{K}$ equals a difference of $1\ \text{°C}$.
\end{formulabox}

\begin{keybox}{Absolute Zero}
At absolute zero ($0\ \text{K} = -273.15\ \text{°C}$):
\begin{itemize}[itemsep=2pt]
  \item Molecules have minimum possible kinetic energy
  \item It is impossible to reach in practice, only to approach asymptotically
  \item All ideal gases would have zero volume (zero pressure at constant volume)
\end{itemize}
\end{keybox}

% ===== SECTION 3: SPECIFIC HEAT CAPACITY AND LATENT HEAT =====
\section{Specific Heat Capacity and Latent Heat}

\begin{definitionbox}{Specific Heat Capacity}
The \textbf{specific heat capacity} $c$ of a substance is the energy required to raise the temperature of \textbf{unit mass} by \textbf{one kelvin} (or one degree Celsius), without change of state.
$$c = \frac{Q}{m\,\Delta T} \qquad \text{units: J\,kg}^{-1}\text{K}^{-1}$$
\end{definitionbox}

\begin{formulabox}{Specific Heat Capacity}
\begin{center}
\Large $Q = mc\Delta T$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$Q$ & = thermal energy transferred (J) \\
$m$ & = mass (kg) \\
$c$ & = specific heat capacity (J\,kg$^{-1}$\,K$^{-1}$) \\
$\Delta T$ & = temperature change (K or °C) \\
\end{tabular}
\end{formulabox}

\begin{definitionbox}{Specific Latent Heat}
The \textbf{specific latent heat} $L$ of a substance is the energy required to change the state of \textbf{unit mass} at constant temperature.
$$L = \frac{Q}{m} \qquad \text{units: J\,kg}^{-1}$$

\begin{itemize}[itemsep=2pt]
  \item \textbf{Specific latent heat of fusion} $L_f$: solid $\rightarrow$ liquid (melting)
  \item \textbf{Specific latent heat of vaporisation} $L_v$: liquid $\rightarrow$ gas (boiling)
\end{itemize}
Note: $L_v > L_f$ for any given substance, since greater work is done separating molecules completely during vaporisation.
\end{definitionbox}

\begin{formulabox}{Specific Latent Heat}
\begin{center}
\Large $Q = mL$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$Q$ & = thermal energy transferred (J) \\
$m$ & = mass (kg) \\
$L$ & = specific latent heat (J\,kg$^{-1}$) \\
\end{tabular}
\end{formulabox}

\begin{warningbox}{Common Mistake}
During a change of state, temperature remains \textbf{constant} even though energy is being supplied. The energy goes into breaking intermolecular bonds, not increasing kinetic energy.
\end{warningbox}

% ===== SECTION 4: THE MOLE AND IDEAL GASES =====
\section{The Mole and Ideal Gases}

\begin{definitionbox}{Amount of Substance}
The \textbf{mole} (mol) is the SI base unit for amount of substance. One mole of any substance contains exactly $N_A = 6.02 \times 10^{23}$ particles (the Avogadro constant).
\end{definitionbox}

\begin{formulabox}{Equation of State for an Ideal Gas}
\begin{center}
\Large $pV = nRT$ \qquad or \qquad $pV = NkT$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$p$ & = pressure (Pa) \\
$V$ & = volume (m$^3$) \\
$n$ & = number of moles (mol) \\
$R$ & = molar gas constant $= 8.31\ \text{J\,mol}^{-1}\text{K}^{-1}$ \\
$T$ & = thermodynamic temperature (K) \\
$N$ & = number of molecules \\
$k$ & = Boltzmann constant $= 1.38 \times 10^{-23}\ \text{J\,K}^{-1}$ \\
\end{tabular}

\medskip
The Boltzmann constant: $k = R/N_A$
\end{formulabox}

\begin{definitionbox}{Ideal Gas}
An \textbf{ideal gas} is one that obeys $pV \propto T$ (where $T$ is thermodynamic temperature) under all conditions. Real gases approximate ideal behaviour at low pressures and high temperatures.
\end{definitionbox}

\begin{keybox}{Gas Laws --- special cases of \texorpdfstring{$pV = nRT$}{pV = nRT}}
\begin{itemize}[itemsep=4pt]
  \item \textbf{Boyle's Law} (constant $T$): $pV = \text{constant}$, so $p_1V_1 = p_2V_2$
  \item \textbf{Charles' Law} (constant $p$): $V/T = \text{constant}$, so $V_1/T_1 = V_2/T_2$
  \item \textbf{Pressure Law} (constant $V$): $p/T = \text{constant}$, so $p_1/T_1 = p_2/T_2$
\end{itemize}
\end{keybox}

\begin{warningbox}{Common Mistake}
Always use \textbf{thermodynamic temperature in kelvin} in gas law calculations --- never degrees Celsius.
\end{warningbox}

% ===== SECTION 5: KINETIC THEORY OF GASES =====
\section{Kinetic Theory of Gases}

\subsection{Assumptions of the Kinetic Model}

\begin{keybox}{Basic Assumptions of Kinetic Theory}
\begin{enumerate}[itemsep=3pt]
  \item The gas contains a \textbf{large number} of molecules moving in \textbf{random directions} with a range of speeds.
  \item The molecules occupy \textbf{negligible volume} compared to the volume of the gas.
  \item All collisions are \textbf{perfectly elastic} (no kinetic energy lost).
  \item \textbf{Intermolecular forces are negligible} except during collisions.
  \item The duration of collisions is \textbf{negligible} compared to the time between collisions.
  \item Molecules obey \textbf{Newton's laws of motion}.
\end{enumerate}
\end{keybox}

\subsection{Pressure from Kinetic Theory}

The pressure exerted by a gas arises from molecules colliding with the container walls. Each collision exerts a force; the average of many such collisions gives a steady pressure.

\begin{formulabox}{Kinetic Theory Equation}
\begin{center}
\Large $pV = \dfrac{1}{3}Nm\langle c^2 \rangle$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$p$ & = pressure (Pa) \\
$V$ & = volume (m$^3$) \\
$N$ & = total number of molecules \\
$m$ & = mass of one molecule (kg) \\
$\langle c^2 \rangle$ & = mean square speed (m$^2$\,s$^{-2}$) \\
\end{tabular}

\medskip
The \textbf{root-mean-square speed}: $c_\text{r.m.s.} = \sqrt{\langle c^2 \rangle}$
\end{formulabox}

\subsection{Molecular Kinetic Energy}

Comparing $pV = \frac{1}{3}Nm\langle c^2\rangle$ with $pV = NkT$:

\begin{formulabox}{Average Translational Kinetic Energy}
\begin{center}
\Large $\langle E_k \rangle = \dfrac{1}{2}m\langle c^2 \rangle = \dfrac{3}{2}kT$
\end{center}

The average kinetic energy of a molecule is \textbf{directly proportional to thermodynamic temperature}. This is a fundamental result.
\end{formulabox}

\begin{keybox}{Key Deductions}
\begin{itemize}[itemsep=3pt]
  \item At absolute zero, molecules have minimum kinetic energy
  \item Molecules of different gases at the same temperature have the \textbf{same average KE}
  \item Heavier molecules move more \textbf{slowly} on average at the same temperature
  \item $c_\text{r.m.s.} \propto \sqrt{T}$
\end{itemize}
\end{keybox}

% ===== SECTION 6: INTERNAL ENERGY =====
\section{Internal Energy}

\begin{definitionbox}{Internal Energy}
The \textbf{internal energy} of a system is the sum of the \textbf{random kinetic energies} and \textbf{potential energies} of all the molecules in the system.

$$U = E_{k,\text{total}} + E_{p,\text{total}}$$

\begin{itemize}[itemsep=2pt]
  \item Internal energy is a \textbf{function of state}.
  \item For an \textbf{ideal gas}: $E_p = 0$, so $U = E_{k,\text{total}}$ only.
  \item A rise in temperature $\Rightarrow$ increase in average KE $\Rightarrow$ increase in $U$.
  \item During a change of state: temperature (and KE) constant; $E_p$ increases as bonds break; $U$ increases.
\end{itemize}
\end{definitionbox}

% ===== SECTION 7: THE FIRST LAW OF THERMODYNAMICS =====
\section{The First Law of Thermodynamics}

\begin{formulabox}{Work Done by/on a Gas}
When a gas changes volume at \textbf{constant pressure}:
\begin{center}
\Large $W = p\,\Delta V$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$W$ & = work done \textbf{by} the gas (J) \\
$p$ & = pressure (Pa) \\
$\Delta V$ & = change in volume (m$^3$) \\
\end{tabular}
\end{formulabox}

\begin{formulabox}{First Law of Thermodynamics}
\begin{center}
\Large $\Delta U = q + W$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$\Delta U$ & = increase in internal energy of the system (J) \\
$q$ & = energy supplied \textbf{to} the system by heating (J) \\
$W$ & = work done \textbf{on} the system (J) \\
\end{tabular}
\end{formulabox}

\begin{warningbox}{Sign Convention}
CIE uses $\Delta U = q + W$ where $W$ is work done \textbf{on} the system. Some textbooks use $\Delta U = q - W$ where $W$ is work done \textbf{by} the system. Always check which convention is being used.
\end{warningbox}

\begin{keybox}{Special Cases of the First Law}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Process} & \textbf{Condition} & \textbf{Result} \\
\midrule
Isothermal & $\Delta T = 0 \Rightarrow \Delta U = 0$ & $q = -W$ \\
Adiabatic  & $q = 0$ & $\Delta U = W$ \\
Isochoric  & $\Delta V = 0 \Rightarrow W = 0$ & $\Delta U = q$ \\
Isobaric   & constant $p$ & $\Delta U = q + p\Delta V$ \\
\bottomrule
\end{tabular}
\end{keybox}

% ===== WORKED EXAMPLES =====
\section{Worked Examples}

\subsection*{Example 1 --- Specific Heat Capacity}

\textbf{Question:} A 2.0 kg block of aluminium ($c = 900\ \text{J\,kg}^{-1}\text{K}^{-1}$) is heated from $20\ \text{°C}$ to $80\ \text{°C}$. Calculate the energy supplied.

\begin{examplebox}{Solution}
$$Q = mc\Delta T = 2.0 \times 900 \times (80-20) = \mathbf{1.08 \times 10^5\ \text{J}}$$
\end{examplebox}

\subsection*{Example 2 --- Ideal Gas Law}

\textbf{Question:} A fixed mass of ideal gas has pressure $1.2\times10^5\ \text{Pa}$ and volume $3.0\times10^{-3}\ \text{m}^3$ at $27\ \text{°C}$. It is heated at constant pressure until its volume doubles. Find the final temperature.

\begin{examplebox}{Solution}
$T_1 = 27 + 273.15 = 300.15\ \text{K}$. At constant pressure, $V \propto T$:
$$T_2 = T_1 \times \frac{V_2}{V_1} = 300.15 \times 2 = \mathbf{600\ \text{K}\ (327\ \text{°C})}$$
\end{examplebox}

\subsection*{Example 3 --- r.m.s.\ Speed}

\textbf{Question:} Calculate the r.m.s.\ speed of nitrogen molecules ($M = 0.028\ \text{kg\,mol}^{-1}$) at $300\ \text{K}$.

\begin{examplebox}{Solution}
Mass of one molecule: $m = 0.028/6.02\times10^{23} = 4.65\times10^{-26}\ \text{kg}$

$$c_\text{r.m.s.} = \sqrt{\frac{3kT}{m}} = \sqrt{\frac{3 \times 1.38\times10^{-23} \times 300}{4.65\times10^{-26}}} = \sqrt{2.67\times10^5} = \mathbf{517\ \text{m\,s}^{-1}}$$
\end{examplebox}

\subsection*{Example 4 --- First Law}

\textbf{Question:} A gas is compressed. $650\ \text{J}$ of work is done on the gas and $200\ \text{J}$ of heat flows out. Find the change in internal energy.

\begin{examplebox}{Solution}
$W = +650\ \text{J}$ (work done \emph{on} gas), \quad $q = -200\ \text{J}$ (heat leaves)
$$\Delta U = q + W = -200 + 650 = \mathbf{+450\ \text{J}}$$
Internal energy \textbf{increases} by 450 J.
\end{examplebox}
