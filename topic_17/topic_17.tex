% ===== SECTION 1: KEY CONCEPTS =====
\section{Core Concepts and Definitions}

\begin{definitionbox}{Oscillatory Motion — Key Terms}
\begin{itemize}[itemsep=2pt]
  \item \textbf{Displacement} $x$: the distance from the equilibrium position, with direction (m).
  \item \textbf{Amplitude} $x_0$: the maximum displacement from equilibrium (m).
  \item \textbf{Period} $T$: the time for one complete oscillation (s).
  \item \textbf{Frequency} $f$: the number of complete oscillations per second (Hz); $f = 1/T$.
  \item \textbf{Angular frequency} $\omega$: $\omega = 2\pi f = 2\pi/T$ (rad s$^{-1}$).
  \item \textbf{Phase difference} $\phi$: the fraction of a cycle by which one oscillation leads or lags another, expressed in radians.
\end{itemize}
\end{definitionbox}

\begin{definitionbox}{Simple Harmonic Motion (SHM)}
An oscillation is \textbf{simple harmonic} if the acceleration of the object is:
\begin{itemize}[itemsep=2pt]
  \item \textbf{proportional} to its displacement from a fixed equilibrium point, and
  \item always directed \textbf{towards} that equilibrium point (opposite to displacement).
\end{itemize}
$$a \propto -x$$
\end{definitionbox}

% ===== SECTION 2: SHM EQUATIONS =====
\section{Equations of Simple Harmonic Motion}

\begin{formulabox}{Defining Equation of SHM}
\begin{center}
\Large $a = -\omega^2 x$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$a$ & = acceleration (m s$^{-2}$) \\
$\omega$ & = angular frequency (rad s$^{-1}$) \\
$x$ & = displacement from equilibrium (m) \\
\end{tabular}
\vspace{0.4em}

The negative sign confirms acceleration is always directed \textbf{opposite} to displacement.
\end{formulabox}

\begin{formulabox}{Displacement; Velocity and Acceleration}
\begin{center}
\large
$x = x_0 \sin\omega t$ \qquad (starting from equilibrium)
\end{center}
\vspace{0.3em}
\begin{center}
\large
$v = v_0 \cos\omega t$ \qquad where $v_0 = \omega x_0$
\end{center}
\vspace{0.3em}
\begin{center}
\large
$v = \pm\,\omega\sqrt{x_0^2 - x^2}$
\end{center}
\vspace{0.4em}
\begin{tabular}{@{}ll@{}}
$x_0$ & = amplitude (m) \\
$v_0$ & = maximum speed $= \omega x_0$ (m s$^{-1}$) \\
$t$ & = time (s) \\
\end{tabular}
\end{formulabox}

\begin{keybox}{Maxima and Minima}
\begin{itemize}[itemsep=2pt]
  \item Acceleration is \textbf{maximum} at maximum displacement ($x = \pm x_0$): \quad $|a_{\max}| = \omega^2 x_0$
  \item Speed is \textbf{maximum} at the equilibrium position ($x = 0$): \quad $v_{\max} = \omega x_0$
  \item Speed is \textbf{zero} at the turning points ($x = \pm x_0$).
\end{itemize}
\end{keybox}

\begin{keybox}{Phase Relationships}
\begin{itemize}[itemsep=2pt]
  \item $v$ leads $x$ by $\tfrac{\pi}{2}$ rad (quarter period ahead).
  \item $a$ leads $v$ by $\tfrac{\pi}{2}$ rad, so $a$ is \textbf{antiphase} ($\pi$ rad ahead) with $x$.
\end{itemize}
\end{keybox}

\begin{warningbox}{Common Mistake}
The equation $x = x_0 \sin\omega t$ assumes the object starts at the \textbf{equilibrium position} at $t = 0$. If it starts at maximum displacement, use $x = x_0 \cos\omega t$ instead. Always check the initial conditions.
\end{warningbox}

% ===== SECTION 3: ENERGY IN SHM =====
\section{Energy in Simple Harmonic Motion}

\begin{formulabox}{Total Energy of a SHM System}
\begin{center}
\Large $E = \tfrac{1}{2}m\omega^2 x_0^2$
\end{center}
\vspace{0.3em}
\begin{tabular}{@{}ll@{}}
$E$ & = total mechanical energy (J) \\
$m$ & = mass of the oscillating object (kg) \\
$\omega$ & = angular frequency (rad s$^{-1}$) \\
$x_0$ & = amplitude (m) \\
\end{tabular}
\end{formulabox}

\begin{keybox}{Energy Interchange During SHM}
\begin{itemize}[itemsep=2pt]
  \item At the \textbf{equilibrium position} ($x = 0$): KE is maximum, PE is zero.
  \item At the \textbf{turning points} ($x = \pm x_0$): KE is zero, PE is maximum.
  \item The \textbf{total energy remains constant} throughout (assuming no damping).
  \item KE $= \tfrac{1}{2}m\omega^2(x_0^2 - x^2)$ \qquad PE $= \tfrac{1}{2}m\omega^2 x^2$
\end{itemize}
\end{keybox}

% ===== SECTION 4: DAMPED OSCILLATIONS =====
\section{Damped and Forced Oscillations}

\subsection{Damping}

\begin{definitionbox}{Damping}
\textbf{Damping} occurs when a resistive force (e.g., air resistance, viscosity) acts on an oscillating system. Energy is removed from the system, causing the amplitude to \textbf{decrease over time}. The frequency is approximately unchanged for light damping.
\end{definitionbox}

\begin{keybox}{Types of Damping}
\begin{itemize}[itemsep=3pt]
  \item \textbf{Light damping}: amplitude decreases gradually over many oscillations; the system oscillates for a long time before coming to rest.
  \item \textbf{Critical damping}: the system returns to equilibrium in the \textbf{shortest possible time} without oscillating. Used in car suspension and door closers.
  \item \textbf{Heavy (overdamping)}: the system returns to equilibrium \textbf{slowly} without oscillating; slower return than critical damping.
\end{itemize}
\end{keybox}

\subsection{Forced Oscillations and Resonance}

\begin{definitionbox}{Forced Oscillations and Natural Frequency}
\begin{itemize}[itemsep=2pt]
  \item Every oscillating system has a \textbf{natural frequency} $f_0$ at which it oscillates freely when displaced and released.
  \item A \textbf{forced oscillation} occurs when a periodic \textbf{driving force} is applied to the system at a \textbf{driving frequency} $f$.
  \item The system oscillates at the \textbf{driving frequency}, not necessarily at its natural frequency.
\end{itemize}
\end{definitionbox}

\begin{definitionbox}{Resonance}
\textbf{Resonance} occurs when the driving frequency equals the natural frequency of the system ($f = f_0$). At resonance:
\begin{itemize}[itemsep=2pt]
  \item The amplitude of oscillation is a \textbf{maximum}.
  \item Energy transfer from the driver to the system is most efficient.
  \item The degree of damping determines how sharp the resonance peak is and the maximum amplitude reached.
\end{itemize}
\end{definitionbox}

\begin{keybox}{Effect of Damping on Resonance}
\begin{itemize}[itemsep=2pt]
  \item More damping $\Rightarrow$ lower and broader resonance peak.
  \item More damping $\Rightarrow$ peak shifts slightly \textbf{below} $f_0$.
  \item Less damping $\Rightarrow$ sharper, taller peak closer to $f_0$.
\end{itemize}
\end{keybox}

\begin{warningbox}{Resonance in Real Life}
Resonance can be \textbf{useful} (e.g.\ MRI scanners, musical instruments, microwave ovens) or \textbf{destructive} (e.g.\ bridges vibrating in wind, buildings in earthquakes). Engineers use damping to control unwanted resonance.
\end{warningbox}

% ===== SECTION 5: EXAMPLES =====
\section{Worked Examples}

\subsection*{Example 1 — Finding Angular Frequency and Max Speed}

\textbf{Question:} A mass oscillates with SHM, amplitude $4.0\ \text{cm}$ and period $0.80\ \text{s}$. Calculate (a) the angular frequency and (b) the maximum speed.

\begin{examplebox}{Solution}
\textbf{(a)} \quad $\omega = \dfrac{2\pi}{T} = \dfrac{2\pi}{0.80} = \mathbf{7.85\ \text{rad s}^{-1}}$

\vspace{0.5em}
\textbf{(b)} \quad $v_{\max} = \omega x_0 = 7.85 \times 0.040 = \mathbf{0.31\ \text{m s}^{-1}}$
\end{examplebox}

\subsection*{Example 2 — Speed at a Given Displacement}

\textbf{Question:} The same mass ($\omega = 7.85\ \text{rad s}^{-1}$, $x_0 = 4.0\ \text{cm}$) has displacement $x = 2.5\ \text{cm}$. Find its speed.

\begin{examplebox}{Solution}
$$v = \omega\sqrt{x_0^2 - x^2} = 7.85\times\sqrt{(0.040)^2 - (0.025)^2}$$
$$v = 7.85\times\sqrt{1.60\times10^{-3} - 6.25\times10^{-4}} = 7.85\times\sqrt{9.75\times10^{-4}}$$
$$v = 7.85\times 0.0312 = \mathbf{0.245\ \text{m s}^{-1}}$$
\end{examplebox}

\subsection*{Example 3 — Total Energy of SHM System}

\textbf{Question:} A $0.20\ \text{kg}$ mass oscillates with $\omega = 7.85\ \text{rad s}^{-1}$ and amplitude $4.0\ \text{cm}$. Calculate the total energy.

\begin{examplebox}{Solution}
$$E = \tfrac{1}{2}m\omega^2 x_0^2 = \tfrac{1}{2}\times 0.20 \times (7.85)^2 \times (0.040)^2$$
$$E = 0.10\times 61.6\times 1.60\times10^{-3} = \mathbf{9.9\times10^{-3}\ \text{J}}$$
\end{examplebox}
