% ── Title page command ───────────────────────────────────────
%  \maketitlepage{Topic number}{Topic title}{Bullet list of contents}
%  Example:
%    \maketitlepage{13}{Gravitational Fields}{
%      \item Newton's Law of Gravitation
%      \item Gravitational Potential
%    }
\newcommand{\maketitlepage}[3]{%
  \thispagestyle{empty}
  \begin{tikzpicture}[remember picture, overlay]
    \fill[darkblue] (current page.north west)
      rectangle ([yshift=-6cm]current page.north east);
    \fill[lightsilver] ([yshift=-6cm]current page.north west)
      rectangle ([yshift=-6.3cm]current page.north east);
    \fill[pinkframe] ([yshift=-6.3cm]current page.north west)
      rectangle ([yshift=-6.55cm]current page.north east);
    \node[anchor=north, text=white, font=\fontsize{16}{20}\selectfont]
      at ([yshift=-1.5cm]current page.north) {CIE International A Level Physics};
    \if\relax\detokenize{#1}\relax\else
    \node[anchor=north, text=white, font=\fontsize{28}{34}\selectfont\bfseries]
      at ([yshift=-3.2cm]current page.north) {Topic #1};
    \fi
  \end{tikzpicture}
  \vspace*{4.5cm}
  \begin{center}
    {\color{darkgray}\fontsize{32}{38}\selectfont\bfseries #2}\\[0.2cm]
    {\color{pinkframe}\rule{0.6\textwidth}{1.5pt}}\\[0.3cm]
    {\color{darkgray}\fontsize{14}{18}\selectfont Revision Booklet}
    \vspace{2.5cm}

    \begin{tcolorbox}[colback=pinklight, colframe=pinkframe,
        width=0.75\textwidth, arc=8pt, boxrule=1.5pt]
      \centering
      \textbf{\color{darkblue}This booklet covers:}\\[0.3cm]
      \begin{itemize}[leftmargin=1.5em, itemsep=2pt]
        #3
      \end{itemize}
    \end{tcolorbox}
  \end{center}
  \newpage
}

    \node[anchor=north, text=white, font=\fontsize{28}{34}\selectfont\bfseries]
      at ([yshift=-3.2cm]current page.north) {Topic #1};
  \end{tikzpicture}
  \vspace*{4.5cm}
  \begin{center}
