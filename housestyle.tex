\newcommand{\maketitlepage}[4]{%

  \thispagestyle{empty}
  % Full-width banner using tikz to break out of margins
  \begin{tikzpicture}[remember picture, overlay]
    \fill[cieprimary]
      ([xshift=0cm, yshift=0cm]current page.north west)
      rectangle
      ([xshift=0cm, yshift=-7.5cm]current page.north east);
  \end{tikzpicture}%
  \vspace*{0.1cm}
  \begin{center}
    {\fontsize{16}{20}\selectfont\bfseries\color{white}
      CIE International A Level Physics (9702)}\\[10pt]
    {\fontsize{24}{30}\selectfont\bfseries\color{white} #1}\\[8pt]
    {\fontsize{34}{40}\selectfont\bfseries\color{white} #2}\\[10pt]
    {\fontsize{15}{20}\selectfont\color{white!85!cieprimary} #3}
  \end{center}
  \vspace{2.8cm}
  % Pink accent rule
  {\color{ciepink}\rule{\textwidth}{4pt}}
  \vspace{0.6cm}
  \begin{center}
    {\large\bfseries\color{cieprimary} Revision Booklet}\\[0.25cm]
    {\normalsize\color{cieprimary}
      Syllabus 9702 \quad|\quad A Level (A2) \quad|\quad 2025--2027}
  \end{center}
  \vspace{0.7cm}
  % Pink outlined rounded box for "This booklet covers"
  \begin{tcolorbox}[
    enhanced,
    colback=white, colframe=ciepink,
    arc=8pt, boxrule=1.5pt,
    left=8pt, right=8pt, top=6pt, bottom=6pt
  ]
    {\bfseries\color{cieprimary} This booklet covers:}
    \vspace{2pt}
    #4
  \end{tcolorbox}
  \end{titlepage}%
}
