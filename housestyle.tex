% ── Title page command ───────────────────────────────────────
%  \maketitlepage{Topic number}{Topic title}{Bullet list of \items}
%  Leave Topic number empty for the main book cover page.
\newcommand{\maketitlepage}[3]{%
  \thispagestyle{empty}
  \begin{tikzpicture}[remember picture, overlay]
    \fill[darkblue] (current page.north west)
      rectangle ([yshift=-6cm]current page.north east);
    \fill[lightsilver] ([yshift=-6cm]current page.north west)
      rectangle ([yshift=-6.3cm]current page.north east);
    \fill[pinkframe] ([yshift=-6.3cm]current page.north west)
      rectangle ([yshift=-6.55cm]current page.north east);
    \node[anchor=north, text=white, font=\fontsize{16}{20}\selectfont]
      at ([yshift=-1.5cm]current page.north) {CIE International A Level Physics};
    \if\relax\detokenize{#1}\relax\else
      \node[anchor=north, text=white, font=\fontsize{28}{34}\selectfont\bfseries]
        at ([yshift=-3.2cm]current page.north) {Topic #1};
    \fi
  \end{tikzpicture}
  \vspace*{4.5cm}
  \begin{center}
    {\color{darkgray}\fontsize{32}{38}\selectfont\bfseries #2}\\[0.2cm]
    {\color{pinkframe}\rule{0.6\textwidth}{1.5pt}}\\[0.3cm]
    {\color{darkgray}\fontsize{14}{18}\selectfont Revision Booklet}
    \vspace{2.5cm}
    \begin{tcolorbox}[colback=pinklight, colframe=pinkframe,
        width=0.75\textwidth, arc=8pt, boxrule=1.5pt]
      \centering
      \textbf{\color{darkblue}This booklet covers:}\\[0.3cm]
      \begin{itemize}[leftmargin=1.5em, itemsep=2pt]
        #3
      \end{itemize}
    \end{tcolorbox}
  \end{center}
  \newpage
}

% ── Chapter opener command ───────────────────────────────────
%  \chapteropener{Topic number (display)}{Topic title}{Bullet list}{label (optional, defaults to #1)}
%
%  In report class (main.tex):   registers a \chapter* for the TOC
%  In article class (standalone): registers via \addcontentsline
%
\newcommand{\chapteropener}[4][]{%
  % #1 = safe label (optional), #2 = topic number display, #3 = title, #4 = bullet items
  % Determine label: use #1 if given, else #2
  \def\topicLabel{#1}\ifx\topicLabel\empty\def\topicLabel{#2}\fi
  \ifstandalone\else\clearpage\fi
  \thispagestyle{empty}
  % TOC registration and hyperlink anchor
  \ifstandalone
    \refstepcounter{section}%
    \addcontentsline{toc}{section}{#3}%
    \setcounter{section}{0}%
  \else
    \chapter*{\texorpdfstring{#3}{#3}}%
    \addcontentsline{toc}{chapter}{\texorpdfstring{#3}{#3}}%
    \phantomsection\label{topic:\topicLabel}%
  \fi
  % Visual banner
  \begin{tikzpicture}[remember picture, overlay]
    \fill[darkblue] (current page.north west)
      rectangle ([yshift=-6cm]current page.north east);
    \fill[lightsilver] ([yshift=-6cm]current page.north west)
      rectangle ([yshift=-6.3cm]current page.north east);
    \fill[pinkframe] ([yshift=-6.3cm]current page.north west)
      rectangle ([yshift=-6.55cm]current page.north east);
    \node[anchor=north, text=white, font=\fontsize{16}{20}\selectfont]
      at ([yshift=-1.5cm]current page.north) {CIE International A Level Physics};
    \node[anchor=north, text=white, font=\fontsize{28}{34}\selectfont\bfseries]
      at ([yshift=-3.2cm]current page.north) {Topic #2};
  \end{tikzpicture}
  \vspace*{4.5cm}
  \begin{center}
    {\color{darkgray}\fontsize{32}{38}\selectfont\bfseries #3}\\[0.2cm]
    {\color{pinkframe}\rule{0.6\textwidth}{1.5pt}}\\[0.3cm]
    {\color{darkgray}\fontsize{14}{18}\selectfont Revision Booklet}
    \vspace{2.5cm}
    \begin{tcolorbox}[colback=pinklight, colframe=pinkframe,
        width=0.75\textwidth, arc=8pt, boxrule=1.5pt]
      \centering
      \textbf{\color{darkblue}This booklet covers:}\\[0.3cm]
      \begin{itemize}[leftmargin=1.5em, itemsep=2pt]
        #4
      \end{itemize}
    \end{tcolorbox}
  \end{center}
  \newpage
}
